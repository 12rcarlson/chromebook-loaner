// Run setupAuthPassword() once from the Apps Script editor after changing
// NEW_STAFF_PASSWORD below. Do not leave the real password in source control.
function setupAuthPassword() {
  const NEW_STAFF_PASSWORD = 'REMOVED';
  if (NEW_STAFF_PASSWORD === 'CHANGE_ME_BEFORE_RUNNING') {
    throw new Error('Edit NEW_STAFF_PASSWORD inside setupAuthPassword() before running it.');
  }
  const props = PropertiesService.getScriptProperties();
  props.setProperty('APP_PASSWORD_HASH', sha256Base64_(NEW_STAFF_PASSWORD));
  if (!props.getProperty('APP_SESSION_SECRET')) {
    props.setProperty('APP_SESSION_SECRET', Utilities.getUuid() + Utilities.getUuid());
  }
  Logger.log('Chromebook Loaner password hash saved. Remove the plaintext password from setupAuthPassword() now.');
}

// ─── Response helper ──────────────────────────────────────────────────────────

function jsonResponse(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}

function parsePayload_(e) {
  if (e.parameter && e.parameter.payload) return JSON.parse(e.parameter.payload);
  if (e.postData && e.postData.contents) return JSON.parse(e.postData.contents);
  throw new Error('No data received');
}

function sha256Base64_(value) {
  const bytes = Utilities.computeDigest(
    Utilities.DigestAlgorithm.SHA_256,
    String(value),
    Utilities.Charset.UTF_8
  );
  return Utilities.base64EncodeWebSafe(bytes);
}

function signToken_(payload) {
  const secret = PropertiesService.getScriptProperties().getProperty('APP_SESSION_SECRET');
  if (!secret) throw new Error('Authentication is not configured. Run setupAuthPassword().');
  const bytes = Utilities.computeHmacSha256Signature(payload, secret);
  return Utilities.base64EncodeWebSafe(bytes);
}

function createToken_() {
  const email = (Session.getActiveUser().getEmail() || '').toLowerCase();
  if (ALLOWED_EMAILS.length && ALLOWED_EMAILS.indexOf(email) === -1) {
    throw new Error('This Google account is not allowed to use this app.');
  }
  const payload = Utilities.base64EncodeWebSafe(JSON.stringify({
    email: email,
    exp: Math.floor(Date.now() / 1000) + AUTH_TOKEN_SECONDS,
    nonce: Utilities.getUuid()
  }));
  return payload + '.' + signToken_(payload);
}

function verifyToken_(token) {
  if (!token || String(token).indexOf('.') === -1) throw new Error('Missing auth token.');
  const parts = String(token).split('.');
  const payload = parts[0];
  const expected = signToken_(payload);
  if (parts[1] !== expected) throw new Error('Invalid auth token.');

  const decoded = JSON.parse(Utilities.newBlob(Utilities.base64DecodeWebSafe(payload)).getDataAsString());
  if (!decoded.exp || decoded.exp < Math.floor(Date.now() / 1000)) throw new Error('Session expired.');
  const email = String(decoded.email || '').toLowerCase();
  if (ALLOWED_EMAILS.length && ALLOWED_EMAILS.indexOf(email) === -1) {
    throw new Error('This Google account is not allowed to use this app.');
  }
  return decoded;
}

function handleLogin_(data) {
  const expected = PropertiesService.getScriptProperties().getProperty('APP_PASSWORD_HASH');
  if (!expected) throw new Error('Authentication is not configured. Run setupAuthPassword().');
  if (sha256Base64_(data.password || '') !== expected) throw new Error('Incorrect password.');
  return jsonResponse({ status: 'ok', token: createToken_(), expiresIn: AUTH_TOKEN_SECONDS });
}

function cleanText_(value, maxLen) {
  const text = String(value == null ? '' : value).replace(/[\u0000-\u0008\u000B\u000C\u000E-\u001F]/g, '').trim();
  return text.slice(0, maxLen || 250);
}

function assertAllowed_(value, allowed, field) {
  if (allowed.indexOf(String(value || '')) === -1) throw new Error('Invalid ' + field + ': ' + value);
}

function assertBulk_(rows, label) {
  if (!Array.isArray(rows)) throw new Error(label + ' must be an array.');
  if (rows.length > MAX_BULK_ROWS) throw new Error(label + ' exceeds the max import size of ' + MAX_BULK_ROWS + '.');
}

function validateCheckout_(d) {
  if (!d || typeof d !== 'object') throw new Error('Checkout payload is required.');
  const record = {
    id              : cleanText_(d.id, 40),
    studentName     : cleanText_(d.studentName, 120),
    grade           : cleanText_(d.grade, 20),
    building        : cleanText_(d.building, 120),
    assetTag        : cleanText_(d.assetTag, 80),
    serial          : cleanText_(d.serial, 120),
    type            : cleanText_(d.type, 40),
    damagedPart     : cleanText_(d.damagedPart, 160),
    originalAssetTag: cleanText_(d.originalAssetTag, 80),
    date            : cleanText_(d.date, 80),
    due             : cleanText_(d.due, 80),
    notes           : cleanText_(d.notes, 1000),
  };
  if (!record.id || !record.studentName || !record.assetTag || !record.type) {
    throw new Error('Checkout requires ID, student name, asset tag, and incident type.');
  }
  assertAllowed_(record.type, ALLOWED_INCIDENT_TYPES, 'incident type');
  assertAllowed_(record.building, ALLOWED_BUILDINGS, 'building');
  return record;
}

function validateLoanerDevice_(d) {
  if (!d || typeof d !== 'object') throw new Error('Device payload is required.');
  const device = {
    assetTag: cleanText_(d.assetTag, 80),
    serial  : cleanText_(d.serial, 120),
    model   : cleanText_(d.model, 160),
    building: cleanText_(d.building, 120),
    status  : cleanText_(d.status || 'Available', 40),
  };
  if (!device.assetTag) throw new Error('Asset tag is required.');
  assertAllowed_(device.status, ALLOWED_LOANER_STATUSES, 'loaner status');
  assertAllowed_(device.building, ALLOWED_BUILDINGS, 'building');
  return device;
}

function validateAllInventoryDevice_(d) {
  if (!d || typeof d !== 'object') throw new Error('Inventory payload is required.');

    model   : cleanText_(d.model, 160),

function validatePart_(p) {
  if (!p || typeof p !== 'object') throw new Error('Part payload is required.');
  const part = {
    id    : cleanText_(p.id, 80),
    type  : cleanText_(p.type, 80),
    mfr   : cleanText_(p.mfr, 80),
    name  : cleanText_(p.name, 180),
    value : Math.max(0, Number(p.value) || 0),
    stock : Math.max(0, Math.floor(Number(p.stock) || 0)),
    fru   : cleanText_(p.fru, 120),
    models: cleanText_(p.models, 500),
    vendor: cleanText_(p.vendor, 180),
  };
  if (!part.name) throw new Error('Part name is required.');
  return part;
}

// ─── Router ───────────────────────────────────────────────────────────────────

function doPost(e) {
  try {
    const data = parsePayload_(e);
    if (data.action === 'login') return handleLogin_(data);
    verifyToken_(data.authToken);

    if (data.action === 'getAll') return getAllData_();
    if (data.action === 'getStudents') return getStudents_(data.forceRefresh === true);

    const lock = LockService.getScriptLock();
    lock.waitLock(10000);
    try {
      logRequest(data);
      dispatch(data);
    } finally {
      lock.releaseLock();
    }
    return jsonResponse({ status: 'ok', action: data.action });
  } catch (err) {
    Logger.log('doPost error: ' + err.message);
    return jsonResponse({ status: 'error', message: err.message });
  }
}

function getStudents_(forceRefresh) {
  const cached = getCachedStudents_();
  if (!forceRefresh && cached && !cached.stale) {
    return jsonResponse({
      status: 'ok',
      students: cached.students,
      cached: true,
      updatedAt: cached.updatedAt
    });
  }

  try {
    const names = fetchStudentNamesFromDirectory_();
    const updatedAt = new Date().toISOString();
    cacheStudents_(names, updatedAt);
    return jsonResponse({ status: 'ok', students: names, cached: false, updatedAt: updatedAt });
  } catch(err) {
    Logger.log('getStudents error: ' + err.message);
    if (cached && cached.students.length) {
      return jsonResponse({
        status: 'ok',
        students: cached.students,
        cached: true,
        stale: true,
        updatedAt: cached.updatedAt,
        message: 'Student directory is temporarily unavailable. Using the last cached list.'
      });
    }
    return jsonResponse({ status: 'error', message: err.message });
  }
}

function fetchStudentNamesFromDirectory_() {
  const names = [];
  STUDENT_OUS.forEach(ou => {
    let pageToken = null;
    do {
      const params = {
        domain     : 'bureauvalley.net',
        query      : "orgUnitPath='" + ou + "'",
        maxResults : 500,
        orderBy    : 'familyName',
        projection : 'basic',
        fields     : 'nextPageToken,users(name)'
      };
      if (pageToken) params.pageToken = pageToken;
      const result = AdminDirectory.Users.list(params);
      (result.users || []).forEach(u => {
        if (u.name && u.name.fullName) names.push(u.name.fullName);
      });
      pageToken = result.nextPageToken;
    } while (pageToken);
  });
  return [...new Set(names)].sort();
}

function getCachedStudents_() {
  const raw = CacheService.getScriptCache().get(STUDENT_CACHE_KEY) || getStoredStudentCache_();
  if (!raw) return null;
  try {
    const cached = JSON.parse(raw);
    const updatedMs = Date.parse(cached.updatedAt || '');
    const ageSeconds = updatedMs ? (Date.now() - updatedMs) / 1000 : STUDENT_CACHE_STALE_SECONDS + 1;
    return {
      students: Array.isArray(cached.students) ? cached.students : [],
      updatedAt: cached.updatedAt || '',
      stale: ageSeconds > STUDENT_CACHE_SECONDS
    };
  } catch(err) {
    Logger.log('getCachedStudents parse error: ' + err.message);
    return null;
  }
}

function cacheStudents_(students, updatedAt) {
  const payload = JSON.stringify({ students: students, updatedAt: updatedAt });
  try {
    CacheService.getScriptCache().put(STUDENT_CACHE_KEY, payload, STUDENT_CACHE_SECONDS);
  } catch(err) {
    Logger.log('cacheStudents CacheService error: ' + err.message);
  }
  storeStudentCache_(payload);
}

function getStoredStudentCache_() {
  const props = PropertiesService.getScriptProperties();
  const direct = props.getProperty(STUDENT_CACHE_KEY);
  if (direct) return direct;

  const count = Number(props.getProperty(STUDENT_CACHE_CHUNK_COUNT_KEY) || 0);
  if (!count) return null;

  let raw = '';
  for (let i = 0; i < count; i++) {
    raw += props.getProperty(STUDENT_CACHE_CHUNK_PREFIX + i) || '';
  }
  return raw || null;
}

function storeStudentCache_(payload) {
  const props = PropertiesService.getScriptProperties();
  try {
    clearStudentCacheChunks_(props);
    if (payload.length <= STUDENT_CACHE_CHUNK_SIZE) {
      props.setProperty(STUDENT_CACHE_KEY, payload);
      return;
    }

    props.deleteProperty(STUDENT_CACHE_KEY);
    const count = Math.ceil(payload.length / STUDENT_CACHE_CHUNK_SIZE);
    for (let i = 0; i < count; i++) {
      props.setProperty(
        STUDENT_CACHE_CHUNK_PREFIX + i,
        payload.slice(i * STUDENT_CACHE_CHUNK_SIZE, (i + 1) * STUDENT_CACHE_CHUNK_SIZE)
      );
    }
    props.setProperty(STUDENT_CACHE_CHUNK_COUNT_KEY, String(count));
  } catch(err) {
    Logger.log('storeStudentCache error: ' + err.message);
  }
}

function clearStudentCacheChunks_(props) {
  const count = Number(props.getProperty(STUDENT_CACHE_CHUNK_COUNT_KEY) || 0);
  for (let i = 0; i < count; i++) {
    props.deleteProperty(STUDENT_CACHE_CHUNK_PREFIX + i);
  }
  props.deleteProperty(STUDENT_CACHE_CHUNK_COUNT_KEY);
}

function getAllData_() {
  try {
    const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
    const checkoutSheet = ss.getSheetByName(CHECKOUT_SHEET);
    let checkouts = [];
    if (checkoutSheet && checkoutSheet.getLastRow() > 1) {
      const rows = checkoutSheet.getRange(2, 1, checkoutSheet.getLastRow() - 1, CHECKOUT_HEADERS.length).getValues();
      checkouts = rows
        .filter(r => r[0] !== '' && r[0] !== null)
        .map(r => ({
          id              : r[0],
          studentName     : String(r[1]  || ''),
          grade           : String(r[2]  || ''),
          building        : String(r[3]  || ''),
          assetTag        : String(r[4]  || ''),
          serial          : String(r[5]  || ''),
          type            : String(r[6]  || ''),
          damagedPart     : String(r[7]  || ''),
          originalAssetTag: String(r[8]  || ''),
          date            : String(r[9]  || ''),
          due             : String(r[10] || ''),
          status          : String(r[11] || ''),
          returnDate      : String(r[12] || ''),
          notes           : String(r[13] || ''),
        }));
    }

    const invSheet = ss.getSheetByName(INVENTORY_SHEET);
    let inventory = [];
    if (invSheet && invSheet.getLastRow() > 1) {
      const rows = invSheet.getRange(2, 1, invSheet.getLastRow() - 1, INVENTORY_HEADERS.length).getValues();
      inventory = rows
        .filter(r => r[0] !== '' && r[0] !== null)
        .map(r => ({
          assetTag: String(r[0] || ''),
          serial  : String(r[1] || ''),
          model   : String(r[2] || ''),
          building: String(r[3] || ''),
          status  : String(r[4] || ''),
        }));
    }

    const allInvSheet = ss.getSheetByName(ALL_INVENTORY_SHEET);
    let allInventory = [];
    if (allInvSheet && allInvSheet.getLastRow() > 1) {
      const rows = allInvSheet.getRange(2, 1, allInvSheet.getLastRow() - 1, ALL_INVENTORY_HEADERS.length).getValues();
      allInventory = rows
        .filter(r => r[0] !== '' && r[0] !== null)
        .map(r => ({
          assetTag: String(r[0] || ''),
          serial  : String(r[1] || ''),
          model   : String(r[2] || ''),
          building: String(r[3] || ''),
          owner   : String(r[4] || ''),
          status  : String(r[5] || ''),
        }));
    }

    const partsSheet = ss.getSheetByName(PARTS_SHEET);
    let parts = [];
    if (partsSheet && partsSheet.getLastRow() > 1) {
      const rows = partsSheet.getRange(2, 1, partsSheet.getLastRow() - 1, PART_ID_COL).getValues();
      parts = rows
        .filter(r => r[2] !== 'TOTAL INVENTORY VALUE' && r[0] !== '' && r[0] !== null)
        .map(r => ({
          type  : String(r[0] || ''),
          mfr   : String(r[1] || ''),
          name  : String(r[2] || ''),
          value : Number(r[3]) || 0,
          stock : Number(r[4]) || 0,
          fru   : String(r[5] || ''),
          models: String(r[6] || ''),
          vendor: String(r[7] || ''),
          id    : String(r[10] || ''),
        }));
    }

    return jsonResponse({
      status      : 'ok',
      checkouts   : checkouts,
      inventory   : inventory,
      allInventory: allInventory,
      parts       : parts,
    });
  } catch(err) {
    Logger.log('getAllData error: ' + err.message);
    return jsonResponse({ status: 'error', message: err.message });
  }
}

function doGet(e) {
  return jsonResponse({ status: 'error', message: 'GET actions are disabled. Use POST with an auth token.' });
}

function dispatch(data) {
  switch (data.action) {
    // Checkouts (always row-level)
    case 'checkout':         appendCheckout(data);                                    break;
    case 'return':           markReturned(data.id, data.returnDate);                  break;
    case 'deleteCheckout':   deleteCheckout(data.id);                                 break;

    // Loaner pool — single record
    case 'upsertDevice':     upsertDevice(data.device);                               break;
    case 'deleteDevice':     deleteSheetRow(data.assetTag, INVENTORY_SHEET, 1,
                               INVENTORY_HEADERS.length, inventoryStatusColor, 5);    break;

    // Loaner pool — bulk CSV import
    case 'syncInventory':    syncInventory(data.devices);                             break;

    // All Inventory — single record
    case 'upsertAllDevice':  upsertAllDevice(data.device);                            break;
    case 'deleteAllDevice':  deleteSheetRow(data.assetTag, ALL_INVENTORY_SHEET, 1,
                               ALL_INVENTORY_HEADERS.length, allInventoryStatusColor, 6); break;

    // All Inventory — bulk CSV import
    case 'syncAllInventory': syncAllInventory(data.devices);                          break;

    // Parts — single record
    case 'upsertPart':       upsertPart(data.part);                                   break;
    case 'deletePart':       deletePart(data.partId);                                 break;

    // Parts — full refresh (exported CSV re-import, or clear-all)
    case 'syncParts':        syncParts(data.parts);                                   break;

    default: throw new Error('Unknown action: ' + data.action);
  }
}

// ─── Checkouts ────────────────────────────────────────────────────────────────

function appendCheckout(d) {
  d = validateCheckout_(d);
  const ss    = SpreadsheetApp.openById(SPREADSHEET_ID);
  const sheet = getOrCreateSheet(ss, CHECKOUT_SHEET);
  ensureHeaders(sheet, CHECKOUT_HEADERS);
  sheet.appendRow([
    d.id, d.studentName || '', d.grade || '', d.building || '',
    d.assetTag || '', d.serial || '', d.type || '', d.damagedPart || '',
    d.originalAssetTag || '', d.date || '', d.due || '',
    'Active', '', d.notes || ''
  ]);
  formatLastRow(sheet);
  conditionalFormat(sheet);
}

function markReturned(id, returnDate) {
  id = cleanText_(id, 40);
  returnDate = cleanText_(returnDate, 80);
  const ss    = SpreadsheetApp.openById(SPREADSHEET_ID);
  const sheet = getOrCreateSheet(ss, CHECKOUT_SHEET);
  const data  = sheet.getDataRange().getValues();
  const statusCol = data[0].indexOf('Status') + 1;
  const returnCol = data[0].indexOf('Return Date') + 1;
  for (let i = 1; i < data.length; i++) {
    if (String(data[i][0]) === String(id)) {
      if (statusCol) sheet.getRange(i + 1, statusCol).setValue('Returned');
      if (returnCol) sheet.getRange(i + 1, returnCol).setValue(returnDate);
      break;
    }
  }
}

function deleteCheckout(id) {
  id = cleanText_(id, 40);
  if (!id) throw new Error('Checkout ID is required.');
  const ss    = SpreadsheetApp.openById(SPREADSHEET_ID);
  const sheet = getOrCreateSheet(ss, CHECKOUT_SHEET);
  const rowNum = findRowByKey(sheet, id, 1);
  if (!rowNum) return;
  sheet.deleteRow(rowNum);
  conditionalFormat(sheet);
  SpreadsheetApp.flush();
}

// ─── Loaner Inventory — targeted upsert ──────────────────────────────────────

function upsertDevice(d) {
  d = validateLoanerDevice_(d);
  const ss    = SpreadsheetApp.openById(SPREADSHEET_ID);
  const sheet = getOrCreateSheet(ss, INVENTORY_SHEET);
  const now   = new Date().toLocaleString('en-US');
  ensureHeaders(sheet, INVENTORY_HEADERS);

  const row = [
    d.assetTag || '', d.serial || '', d.model || '',
    d.building || '', d.status || '', now
  ];

  const rowNum = findRowByKey(sheet, d.assetTag, 1);
  if (rowNum) {
    sheet.getRange(rowNum, 1, 1, INVENTORY_HEADERS.length).setValues([row]);
    styleInventoryRow(sheet, rowNum, d.status);
  } else {
    sheet.appendRow(row);
    styleInventoryRow(sheet, sheet.getLastRow(), d.status);
  }
  SpreadsheetApp.flush();
}

function styleInventoryRow(sheet, rowNum, status) {
  const bg = rowNum % 2 === 0 ? '#f8f7f4' : null;
  sheet.getRange(rowNum, 1, 1, 4).setBackground(bg);
  sheet.getRange(rowNum, 6, 1, 1).setBackground(bg);
  sheet.getRange(rowNum, 5, 1, 1).setBackground(inventoryStatusColor(status));
}

function inventoryStatusColor(status) {
  switch (String(status)) {
    case 'Available':    return '#e1f5ee';
    case 'On Loan':      return '#e6f1fb';
    case 'Loaner':       return '#e6f1fb';
    case 'Daily Loaner': return '#e6f1fb';
    case 'Needs Repair': return '#fcebeb';
    default:             return '#f1efe8';
  }
}

function styleAllInventoryRow(sheet, rowNum, status) {
  const bg = rowNum % 2 === 0 ? '#f8f7f4' : null;
  sheet.getRange(rowNum, 1, 1, 5).setBackground(bg);
  sheet.getRange(rowNum, 7, 1, 1).setBackground(bg);
  sheet.getRange(rowNum, 6, 1, 1).setBackground(allInventoryStatusColor(status));
}

function allInventoryStatusColor(status) {
  switch (String(status)) {
    case 'Assigned':     return '#e6f1fb';
    case 'Unassigned':   return '#e1f5ee';
    case 'Loaner':       return '#faeeda';
    case 'Daily Loaner': return '#faeeda';
    case 'Needs Repair': return '#fcebeb';
    default:             return '#f1efe8';
  }
}

// ─── Shared row delete (Inventory + All Inventory) ────────────────────────────

function deleteSheetRow(key, sheetName, keyCol, numCols, statusColorFn, statusCol) {
  key = cleanText_(key, 80);
  if (!key) throw new Error('Delete key is required.');
  const ss    = SpreadsheetApp.openById(SPREADSHEET_ID);
  const sheet = getOrCreateSheet(ss, sheetName);
  const rowNum = findRowByKey(sheet, key, keyCol);
  if (!rowNum) return;
  sheet.deleteRow(rowNum);
  restripeFrom(sheet, rowNum, numCols, statusCol, statusColorFn,
    sheetName === PARTS_SHEET ? '#ffffff' : null);
  SpreadsheetApp.flush();
}

// ─── Parts — targeted upsert / delete ────────────────────────────────────────

function upsertPart(p) {
  p = validatePart_(p);
  const ss    = SpreadsheetApp.openById(SPREADSHEET_ID);
  const sheet = getOrCreateSheet(ss, PARTS_SHEET);
  const now   = new Date().toLocaleString('en-US');
  ensureHeaders(sheet, PARTS_HEADERS);

  const value    = Number(p.value) || 0;
  const stock    = Number(p.stock) || 0;
  const lineTotal = Number((value * stock).toFixed(2));

  const row = [
    p.type || '', p.mfr || '', p.name || '',
    value, stock, p.fru || '', p.models || '', p.vendor || '',
    lineTotal, now
  ];

  let rowNum = p.id ? findRowByKey(sheet, p.id, PART_ID_COL) : null;
  if (rowNum) {
    sheet.getRange(rowNum, 1, 1, PARTS_HEADERS.length).setValues([row]);
  } else {
    sheet.appendRow(row);
    rowNum = sheet.getLastRow();
    sheet.getRange(rowNum, PART_ID_COL).setValue(p.id || '');
  }

  stylePartsRow(sheet, rowNum, stock);
  sheet.getRange(rowNum, 4, 1, 1).setNumberFormat('$#,##0.00');
  sheet.getRange(rowNum, 9, 1, 1).setNumberFormat('$#,##0.00');
  updatePartsTotalsRow(sheet);
  SpreadsheetApp.flush();
}

function deletePart(partId) {
  partId = cleanText_(partId, 80);
  if (!partId) throw new Error('Part ID is required.');
  const ss    = SpreadsheetApp.openById(SPREADSHEET_ID);
  const sheet = getOrCreateSheet(ss, PARTS_SHEET);
  const rowNum = findRowByKey(sheet, partId, PART_ID_COL);
  if (!rowNum) return;
  sheet.deleteRow(rowNum);
  restripeFrom(sheet, rowNum, PARTS_HEADERS.length, 5, stockStatusColor, '#ffffff');
  updatePartsTotalsRow(sheet);
  SpreadsheetApp.flush();
}

function stylePartsRow(sheet, rowNum, stock) {
  const bg = rowNum % 2 === 0 ? '#f8f7f4' : '#ffffff';
  sheet.getRange(rowNum, 1, 1, PARTS_HEADERS.length).setBackground(bg);
  sheet.getRange(rowNum, 5, 1, 1).setBackground(stockStatusColor(stock));
}

function stockStatusColor(stock) {
  const s = Number(stock) || 0;
  if (s === 0)  return '#fcebeb';
  if (s <= 2)   return '#faeeda';
  return '#e1f5ee';
}

function updatePartsTotalsRow(sheet) {
  const lastRow = sheet.getLastRow();
  if (lastRow < 2) return;
  const colC = sheet.getRange(2, 3, lastRow - 1, 1).getValues();
  for (let i = colC.length - 1; i >= 0; i--) {
    if (String(colC[i][0]).trim() === 'TOTAL INVENTORY VALUE') {
      sheet.deleteRow(i + 2);
      break;
    }
  }
  const dataRows = sheet.getLastRow();
  const totalRow = dataRows + 1;
  sheet.getRange(totalRow, 3).setValue('TOTAL INVENTORY VALUE').setFontWeight('bold');
  sheet.getRange(totalRow, 9)
    .setFormula(`=SUM(I2:I${dataRows})`)
    .setNumberFormat('$#,##0.00')
    .setFontWeight('bold')
    .setBackground('#1a56a0')
    .setFontColor('#ffffff');
}

// ─── Utility helpers ──────────────────────────────────────────────────────────

function findRowByKey(sheet, key, colIndex) {
  const lastRow = sheet.getLastRow();
  if (lastRow < 2) return null;
  const vals = sheet.getRange(2, colIndex, lastRow - 1, 1).getValues();
  for (let i = 0; i < vals.length; i++) {
    if (String(vals[i][0]).trim() === String(key).trim()) return i + 2;
  }
  return null;
}

function restripeFrom(sheet, fromRow, numCols, statusCol, statusColorFn, oddBg) {
  const lastRow = sheet.getLastRow();
  if (fromRow > lastRow) return;
  const count = lastRow - fromRow + 1;
  const vals  = sheet.getRange(fromRow, 1, count, numCols).getValues();
  for (let i = 0; i < count; i++) {
    const rowNum = fromRow + i;
    const bg = rowNum % 2 === 0 ? '#f8f7f4' : (oddBg || null);
    sheet.getRange(rowNum, 1, 1, numCols).setBackground(bg);
    const statusVal = vals[i][statusCol - 1];
    sheet.getRange(rowNum, statusCol, 1, 1).setBackground(statusColorFn(statusVal));
  }
}

function clearDataRows(sheet, numCols) {
  const lastRow = sheet.getLastRow();
  if (lastRow > 1) {
    sheet.getRange(2, 1, lastRow - 1, numCols).clearContent().setBackground(null);
  }
}

function logRequest(data) {

  const sheet = getOrCreateSheet(ss, LOG_SHEET);
  ensureHeaders(sheet, ['Timestamp', 'Action', 'Payload']);
  const safe = Object.assign({}, data);
  delete safe.authToken;
  if (safe.studentName) safe.studentName = '[redacted]';
  if (safe.password) safe.password = '[redacted]';
  sheet.appendRow([new Date().toISOString(), data.action || 'unknown', JSON.stringify(safe)]);
}

// ─── Formatting helpers ───────────────────────────────────────────────────────

function ensureHeaders(sheet, headers) {
  const hasRows     = sheet.getLastRow() > 0;
  const firstRow    = hasRows ? sheet.getRange(1, 1, 1, headers.length).getValues()[0] : [];
  const headersMatch = headers.every((h, i) => firstRow[i] === h);
  if (!headersMatch) {
    const r = sheet.getRange(1, 1, 1, headers.length);
    r.setValues([headers]).setFontWeight('bold').setBackground('#1a56a0').setFontColor('#ffffff');
    sheet.setFrozenRows(1);
  }
}

function formatLastRow(sheet) {
  const row = sheet.getLastRow();
  sheet.getRange(row, 1, 1, sheet.getLastColumn())
    .setVerticalAlignment('middle')
    .setBackground(row % 2 === 0 ? '#f8f7f4' : null);
}

function conditionalFormat(sheet) {
  const range = sheet.getRange('L2:L1000');
  const active   = SpreadsheetApp.newConditionalFormatRule().whenTextEqualTo('Active')
    .setBackground('#e1f5ee').setFontColor('#085041').setRanges([range]).build();
  const returned = SpreadsheetApp.newConditionalFormatRule().whenTextEqualTo('Returned')
    .setBackground('#f1efe8').setFontColor('#5f5e5a').setRanges([range]).build();
  const existing = sheet.getConditionalFormatRules().filter(r => {
    const bc = r.getBooleanCondition();
    if (!bc) return true;
    const v = bc.getCriteriaValues()[0];
    return v !== 'Active' && v !== 'Returned';
  });
  sheet.setConditionalFormatRules([...existing, active, returned]);
}

function getOrCreateSheet(ss, name) {
  return ss.getSheetByName(name) || ss.insertSheet(name);
}
